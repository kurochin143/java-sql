SELECT COUNT(o.OrderId) as NumbersOfOrders, c.CustomerName
FROM Orders o
LEFT JOIN Customers c
ON o.CustomerId = c.CustomerId
GROUP BY o.CustomerId

SELECT COUNT(o.OrderId) as NumbersOfOrders, c.CustomerName
FROM Orders o
LEFT JOIN Customers c
ON o.CustomerId = c.CustomerId
GROUP BY o.CustomerId
ORDER BY NumbersOfOrders DESC;

SELECT COUNT(o.OrderId) as NumbersOfOrders, c.City
FROM Orders o
LEFT JOIN Customers c
ON o.CustomerId = c.CustomerId
GROUP BY c.City;
